WIN关闭一个进程

import ctypes
def kill(pid):
    """kill function for Win32"""
    kernel32 = ctypes.windll.kernel32
    handle = kernel32.OpenProcess(1, 0, pid)

    #使用termina函数结束进程
    return (0 != kernel32.TerminateProcess(handle, 0))
